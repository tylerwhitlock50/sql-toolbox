Sub check_lines()

Dim objShell
Set objShell = CreateObject("WScript.Shell")
Set ORDERLINES = Lines.Value
LineCount = ORDERLINES.Count

If LineCount = 0 Then Exit Sub

Foundreturnpart = False

For i = 0 To (LineCount - 1)
    Set oline = ORDERLINES(i).Value
    If InStr(1, oline("PART_ID"), "RETURN") > 0 Then
        Foundreturnpart = True
        Exit For
    End If
    Next
    
If Foundreturnpart Then
 objShell.Run "https://youtu.be/2ZOIpsxfzxw?si=hutjyBxzn0DKe1Bz", 1, False



' Initial prompt

' First follow-up
response = MsgBox("Hmm, it seems like you might not have gotten the memo. This really isn't recommended. Still want to proceed?", vbYesNo + vbQuestion, "Second Thoughts?")
If response = vbNo Then Exit Sub

' Second follow-up
response = MsgBox("Okay, now we're entering the territory of 'I told you so'. Are you *really* sure? Like, really-really sure?", vbYesNo + vbExclamation, "Seriously?")
If response = vbNo Then Exit Sub

' Third follow-up
response = MsgBox("We're not mad, just disappointed. But hey, if you insist on clicking things you shouldn't... Last chance to turn back!", vbYesNo + vbInformation, "Last Chance")

' Final message
