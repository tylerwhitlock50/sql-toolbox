select CUSTOMER.ID , CUSTOMER.NAME , CUSTOMER.ADDR_1 , CUSTOMER.ADDR_2 , CUSTOMER.ADDR_3 , CUSTOMER.CITY , CUSTOMER.STATE , CUSTOMER.ZIPCODE , CUSTOMER.COUNTRY , CUSTOMER.SALESREP_ID , CUSTOMER.TERRITORY , CUSTOMER.TAX_ID_NUMBER , CUSTOMER.TERMS_DESCRIPTION , CUSTOMER.LAST_ORDER_DATE , CUSTOMER.OPEN_DATE , CUSTOMER.USER_4 as  FFL_Number , CUSTOMER.USER_5 as  FFL_Expiration , CUSTOMER.WEB_URL , CUSTOMER_ENTITY.CREDIT_STATUS , CUSTOMER_CARMS.CUSTOMER_GROUP , CUSTOMER_CARMS.CUSTOMER_SUB_GROUP , CUSTOMER_CARMS.CUSTOMER_NAME from VECA.dbo.CUSTOMER CUSTOMER inner join VECA.dbo.CUSTOMER_CARMS CUSTOMER_CARMS on customer_Carms.ID = Customer.ID inner join VECA.dbo.CUSTOMER_ENTITY CUSTOMER_ENTITY on CUSTOMER_ENTITY.CUSTOMER_ID = customer.ID where  CUSTOMER.LAST_ORDER_DATE > '2017-12-31' and customer.active_flag = 'Y'
